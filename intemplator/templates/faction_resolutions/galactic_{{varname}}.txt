resolution_establish_galatic_{{varname}} = {
	icon = "GFX_resolution_united_front"

	resources = {
		category = resolutions
		cost = {
			influence = @resolution_cost_t2
		}
	}
	target = no
	level = 1

	modifier = {
		resolution_force_establish_galatic_{{varname}}_static_modifier = 1
	}

	effect = {
		custom_tooltip = resolution_establish_galatic_{{varname}}_tooltip
	}

	fail_effects = {
	}

	allow = {
		custom_tooltip = {
			fail_text = "resolution_already_active"
			NOR = {
				is_active_resolution = "resolution_establish_galatic_{{varname}}"
			}			
		}
	}

	potential = {
		# Only available to countries that have an ethic
		OR = {
			{% if isMilitarist > 0 %}
			has_ethic = ethic_militarist
			has_ethic = ethic_fanatic_militarist
			{% endif %}

			{% if isXenophobe > 0 %}
			has_ethic = ethic_xenophobe
			has_ethic = ethic_fanatic_xenophobe
			{% endif %}

			{% if isEgalitarian > 0 %}
			has_ethic = ethic_egalitarian
			has_ethic = ethic_fanatic_egalitarian
			{% endif %}

			{% if isMaterialist > 0 %}
			has_ethic = ethic_materialist
			has_ethic = ethic_fanatic_materialist
			{% endif %}

			{% if isPacifist > 0 %}
			has_ethic = ethic_pacifist
			has_ethic = ethic_fanatic_pacifist
			{% endif %}

			{% if isXenophile > 0 %}
			has_ethic = ethic_xenophile
			has_ethic = ethic_fanatic_xenophile
			{% endif %}

			{% if isAuthoritarian > 0 %}
			has_ethic = ethic_authoritarian
			has_ethic = ethic_fanatic_authoritarian
			{% endif %}

			{% if isSpiritualist > 0 %}
			has_ethic = ethic_spiritualist
			has_ethic = ethic_fanatic_spiritualist
			{% endif %}
		}
	}

	ai_weight = { 
		base = @resolution_weight_disliked

        {% if isMilitarist > 1 %}
        modifier = {
			factor = 0
			is_pacifist = yes
			desc = ethic_pacifist
		}
		modifier = {
			factor = 2
			is_militarist = yes
			desc = ethic_militarist
		}
        {% endif %}

        {% if isPacifist > 1 %}
        modifier = {
			factor = 0
			is_militarist = yes
			desc = ethic_militarist
		}
		modifier = {
			factor = 2
			is_pacifist = yes
			desc = ethic_pacifist
		}
        {% endif %}

        {% if isEgalitarian > 1 %}
        modifier = {
			factor = 0
			is_authoritarian = yes
			desc = ethic_authoritarian
		}
		modifier = {
			factor = 2
			is_egalitarian = yes
			desc = ethic_egalitarian
		}
        {% endif %}

        {% if isAuthoritarian > 1 %}
        modifier = {
			factor = 0
			is_egalitarian = yes
			desc = ethic_egalitarian
		}
		modifier = {
			factor = 2
			is_authoritarian = yes
			desc = ethic_authoritarian
		}
        {% endif %}

        {% if isMaterialist > 1 %}
		modifier = {
			factor = 2
			is_materialist = yes
			desc = ethic_materialist
		}
        modifier = {
			factor = 0
			is_spiritualist = yes
			desc = ethic_spiritualist
		}
        {% endif %}

        {% if isSpiritualist > 1 %}
        modifier = {
			factor = 0
			is_materialist = yes
			desc = ethic_materialist
		}
		modifier = {
			factor = 2
			is_spiritualist = yes
			desc = ethic_spiritualist
		}
        {% endif %}

        {% if isXenophobe > 1 %}
		modifier = {
			factor = 2
			is_xenophobe = yes
			desc = ethic_xenophobe
		}
        modifier = {
			factor = 0
			is_xenophile = yes
			desc = ethic_xenophile
		}
        {% endif %}

        {% if isXenophile > 1 %}
		modifier = {
			factor = 2
			is_xenophile = yes
			desc = ethic_xenophile
		}
        modifier = {
			factor = 0
			is_xenophobe = yes
			desc = ethic_xenophobe
		}
        {% endif %}

		modifier = {
			factor = 0
			is_gestalt = yes
			desc = ethic_gestalt_consciousness
		}
	}  
}
