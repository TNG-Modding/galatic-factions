fanatic_spiritualist_pacifist_faction = {
	election_header = "GFX_faction_header_yellow"
	guiding_ethic = ethic_militarist #only pops of this ethic can join the faction

	unique = yes
	support_multiplier = 1

	resources = {
		category = pop_factions
		produces = {
			unity = 0.5
		}
	}

	parameters = {
		empire = {
			type = country
			valid_objects = {
				is_same_value = root
			}
		}
	}

	is_potential = {
		AND = { 
			# is_faction_allowed = yes

			OR = {
				has_modifier = resolution_force_establish_galatic_fanatic_spiritualist_pacifist_faction_static_modifier
				AND = {
					
					
					
					
					
					
					
					
					
					has_ethic = ethic_pacifist
					
					
					
					
					has_ethic = ethic_fanatic_spiritualist
					
				}
			}
			
		}
	}
	can_join_pre_triggers = {
		is_enslaved = no
		is_being_purged = no
	}

	can_join_faction = {
		is_shackled_robot = no
		NOT = { has_trait = trait_nerve_stapled }
		OR = {
			
			
			
			
			has_ethic = ethic_pacifist
			
			
			has_ethic = ethic_spiritualist
		}
	}

	attraction = {
		base = 100

		modifier = {
			factor = 1.25
			exists = from
			from = {
				exists = leader
				leader = { leader_class = ruler }
			}
		}

		modifier = {
			factor = 0.75
			exists = from
			from = {
				NOT = { exists = leader }
			}
		}

		modifier = {
			factor = 1.25
			has_citizenship_rights = yes
		}

		
		# Fanatic factions are enticing if we are also fanatic, this keeps our competative if we are fanatic, 
		# but makes matching fanatic ethics also competative
		modifier = {
			factor = 1.25

			
			
			
			
			
			
			
			has_ethic = ethic_fanatic_spiritualist
		}
		
	}

	leader = {
		base = 100

		modifier = {
			factor = 0.0
			is_event_leader = yes
		}
		modifier = {
			factor = 0.25
			leader_class = ruler
		}
	}

	on_create = {
		# pop_faction_event = { id = faction_spawn.1 }
	}

	actions = {
		
	}
}
