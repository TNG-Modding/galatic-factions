fanatic_pacifist_xenophile_faction = {
	election_header = "GFX_faction_header_yellow"
	guiding_ethic = ethic_militarist #only pops of this ethic can join the faction

	unique = yes
	support_multiplier = 1

	resources = {
		category = pop_factions
		produces = {
			unity = 0.5
		}
	}

	parameters = {
		empire = {
			type = country
			valid_objects = {
				is_same_value = root
			}
		}
	}


	is_potential = {
		AND = { 
			is_faction_allowed = yes
			
			
			
			
			
			
			
			
			has_ethic = ethic_fanatic_pacifist
			
			
			has_ethic = ethic_xenophile
			
			
			
			
		}
	}
	can_join_pre_triggers = {
		is_enslaved = no
		is_being_purged = no
	}

	can_join_faction = {
		is_shackled_robot = no
		NOT = { has_trait = trait_nerve_stapled }
		OR = {
			
			
			
			
			has_ethic = ethic_pacifist
			has_ethic = ethic_xenophile
			
			
		}
	}

	attraction = {
		base = 100

		modifier = {
			factor = 1.25
			exists = from
			from = {
				exists = leader
				leader = { leader_class = ruler }
			}
		}

		modifier = {
			factor = 0.75
			exists = from
			from = {
				NOT = { exists = leader }
			}
		}

		modifier = {
			factor = 1.25
			has_citizenship_rights = yes
		}

		
		modifier = {
			factor = 1.25

			OR = {
				
				
				
				
				has_ethic = ethic_fanatic_pacifist
				
				
				
			}
		}
		
	}

	leader = {
		base = 100

		modifier = {
			factor = 0.0
			is_event_leader = yes
		}
		modifier = {
			factor = 0.25
			leader_class = ruler
		}
	}

	on_create = {
		# pop_faction_event = { id = faction_spawn.1 }
	}

	actions = {
		
	}
}
